CREATE TABLE CLIENTE(
	CLIENTEID INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    NOME VARCHAR(35) NOT NULL
);

CREATE TABLE TELEFONE(
	TELEFONEID INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    NUMERO VARCHAR(35) NOT NULL,
    CLIENTE_ID INT NOT NULL,
    FOREIGN KEY (CLIENTE_ID)
    REFERENCES CLIENTE(CLIENTEID)
);  

CREATE TABLE CARRO(
	CARROID INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    MARCA VARCHAR(20) NOT NULL,
    CLIENTE_ID INT NOT NULL,
    FOREIGN KEY (CLIENTE_ID)
    REFERENCES CLIENTE(CLIENTEID)
)

CREATE TABLE COR(
	CORID INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    COR VARCHAR(20) NOT NULL,
    CARRO_ID INT NOT NULL,
    FOREIGN KEY (CARRO_ID)
    REFERENCES CARRO(CARROID)
);
